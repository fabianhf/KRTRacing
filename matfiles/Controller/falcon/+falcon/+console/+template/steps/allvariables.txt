function vars = allvariables()
    % Load all variables from the 'vars' subfolder into one struct.

    % The variable providers are in a subdirectory. We need
    % to load it before we can use the functions in there.
    addpath vars

    vars = struct( ...
        'states', states(), ...
        'outputs', outputs(), ...
        'controls', controls(), ...
        'parameters', parameters() ...
    );

end
